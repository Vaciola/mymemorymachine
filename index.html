    <script>
        // --- 核心状态 ---
        let currentDays = 1;
        let rotation = 0;
        const knob = document.getElementById('knob');
        const stages = [1, 2, 4, 7, 15, 30];

        // --- 旋钮逻辑 ---
        let isDragging = false;
        let lastAngle = 0;

        const getAngle = (e) => {
            const rect = knob.getBoundingClientRect();
            const center = { x: rect.left + rect.width / 2, y: rect.top + rect.height / 2 };
            const client = e.touches ? e.touches[0] : e;
            return Math.atan2(client.clientY - center.y, client.clientX - center.x) * 180 / Math.PI;
        };

        const handleStart = (e) => { isDragging = true; lastAngle = getAngle(e); e.preventDefault(); };
        const handleMove = (e) => {
            if (!isDragging) return;
            const angle = getAngle(e);
            rotation += (angle - lastAngle);
            knob.style.transform = `rotate(${rotation}deg)`;
            lastAngle = angle;
           
            // 计算艾宾浩斯天数 (简化版)
            let index = Math.abs(Math.floor(rotation / 60) % stages.length);
            currentDays = stages[index];
            document.getElementById('day-val').innerText = currentDays;
            updateNextDate();
        };

        knob.addEventListener('mousedown', handleStart);
        window.addEventListener('mousemove', handleMove);
        window.addEventListener('mouseup', () => isDragging = false);
        knob.addEventListener('touchstart', handleStart, {passive: false});
        window.addEventListener('touchmove', handleMove, {passive: false});
        window.addEventListener('touchend', () => isDragging = false);

        // --- 功能逻辑 ---
        function updateNextDate() {
            const d = new Date();
            d.setDate(d.getDate() + currentDays);
            document.getElementById('next-date').innerText = d.toISOString().split('T')[0];
        }

        function mockInput(type) {
            const display = document.getElementById('quest-display');
            display.innerHTML = `> [${type}] 模式启动...<br>> 正在捕捉数据...<br>> 请在机器上方操作。`;
            setTimeout(() => {
                display.innerHTML = `> 成功捕捉到新题目!<br>> 类型: 数学错题<br>> 建议复习周期: ${currentDays}天`;
            }, 1000);
        }

        function saveErrorNote() {
            const note = {
                id: Date.now(),
                content: "新录入的错题数据 #" + Math.floor(Math.random()*100),
                interval: currentDays,
                nextReview: document.getElementById('next-date').innerText
            };
           
            // 同步到本地存储 (模拟云端)
            let history = JSON.parse(localStorage.getItem('notes') || '[]');
            history.unshift(note);
            localStorage.setItem('notes', JSON.stringify(history));
            renderList();
            alert("同步成功！已计入艾宾浩斯计划。");
        }

        function renderList() {
            const container = document.getElementById('list-container');
            const history = JSON.parse(localStorage.getItem('notes') || '[]');
            container.innerHTML = history.map(n => `
                <div class="panel p-3 rounded-lg text-xs flex justify-between items-center border border-zinc-700">
                    <div>
                        <div class="text-zinc-300">${n.content}</div>
                        <div class="text-[8px] text-zinc-500 mt-1">NEXT: ${n.nextReview}</div>
                    </div>
                    <div class="bg-orange-900/30 text-orange-500 px-2 py-1 rounded">${n.interval}D</div>
                </div>
            `).join('');
        }
